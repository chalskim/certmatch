        let currentProposalId = null;

        // 토스트 메시지 표시 함수
        function showToast(message) {
            const toast = document.getElementById('toast');
            toast.textContent = message;
            toast.classList.add('show');
            
            setTimeout(() => {
                toast.classList.remove('show');
            }, 3000);
        }
        
        // 탭 전환 함수
        function showTab(tabName) {
            // 모든 탭 버튼과 콘텐츠 비활성화
            document.querySelectorAll('.tab-button').forEach(btn => btn.classList.remove('active'));
            document.querySelectorAll('.tab-content').forEach(content => content.classList.remove('active'));
            
            // 선택된 탭 활성화
            document.querySelector(`[data-tab="${tabName}"]`).classList.add('active');
            document.getElementById(`${tabName}`).classList.add('active');
        }
        
        // 카드 생성 함수
        function createCard(item, type) {
            let statusClass = '';
            let statusText = '';
            
            if (type === 'proposal') {
                statusClass = `status-${item.status}`;
                statusText = { pending: '대기', accepted: '수락', rejected: '거절', completed: '완료' }[item.status];
            } else if (type === 'project') {
                statusClass = item.status === 'ongoing' ? 'status-ongoing' : 'status-completed';
                statusText = item.status === 'ongoing' ? '진행중' : '완료';
            }
            
            return `
                <div class="card" onclick="${type === 'proposal' ? `showProposalDetail(${item.id})` : ''}">
                    <div class="card-header">
                        <div class="card-title">${item.title}</div>
                        <div class="card-status ${statusClass}">${statusText}</div>
                    </div>
                    <div class="card-company">${item.company}</div>
                    <div class="card-meta">
                        <span>${item.date || `${item.startDate} ~ ${item.endDate}`}</span>
                    </div>
                </div>
            `;
        }
        
        // 리뷰 카드 생성 함수
        function createReviewCard(review) {
            const stars = '★'.repeat(review.rating) + '☆'.repeat(5 - review.rating);
            return `
                <div class="review-card">
                    <div class="review-header">
                        <div class="reviewer-info">
                            <div class="reviewer-avatar"><i class="fas fa-user"></i></div>
                            <div>
                                <div class="reviewer-name">${review.name}</div>
                                <div class="review-company">${review.company}</div>
                            </div>
                        </div>
                        <div class="review-rating">${stars}</div>
                    </div>
                    <div class="review-content">${review.content}</div>
                    <div class="review-date">${review.date}</div>
                </div>
            `;
        }
        
        // 프로젝트 필터링 함수
        function filterProjects(filter, button) {
            currentProjectFilter = filter;
            
            // 필터 버튼 활성화 상태 변경
            button.parentElement.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            button.classList.add('active');
            
            // 프로젝트 필터링
            let filteredProjects = mockData.projects;
            if (filter !== 'all') {
                filteredProjects = mockData.projects.filter(p => p.status === filter);
            }
            
            // 현재 탭에 따라 다른 리스트 업데이트
            const activeTab = document.querySelector('.tab-content.active').id;
            const listId = activeTab === 'dashboard' ? 'allProjectsList' : 'projectsTabList';
            
            const listElement = document.getElementById(listId);
            listElement.innerHTML = filteredProjects.length > 0
                ? filteredProjects.map(p => createCard(p, 'project')).join('')
                : '<div class="empty-state"><i class="fas fa-project-diagram"></i><p>해당 상태의 프로젝트가 없습니다.</p></div>';
        }
        
        // 제안 필터링 함수
        function filterProposals(filter, button) {
            currentProposalFilter = filter;
            
            // 필터 버튼 활성화 상태 변경
            button.parentElement.querySelectorAll('.filter-btn').forEach(btn => btn.classList.remove('active'));
            button.classList.add('active');
            
            // 제안 필터링
            let filteredProposals = mockData.proposals;
            if (filter !== 'all') {
                filteredProposals = mockData.proposals.filter(p => p.status === filter);
            }
            
            // 현재 탭에 따라 다른 리스트 업데이트
            const activeTab = document.querySelector('.tab-content.active').id;
            const listId = activeTab === 'dashboard' ? 'allProposalsList' : 'proposalsTabList';
            
            const listElement = document.getElementById(listId);
            listElement.innerHTML = filteredProposals.length > 0
                ? filteredProposals.map(p => createCard(p, 'proposal')).join('')
                : '<div class="empty-state"><i class="fas fa-envelope"></i><p>해당 상태의 제안이 없습니다.</p></div>';
        }
        
        // 대시보드 렌더링
        function renderDashboard() {
            // 통계 카드 업데이트
            document.querySelector('.stat-card:nth-child(1) .stat-value').textContent = mockData.projects.length;
            document.querySelector('.stat-card:nth-child(2) .stat-value').textContent = mockData.proposals.length;
            
            // 초기에는 전체 목록 표시
            filterProjects('all', document.querySelector('#allProjectsList').parentElement.querySelector('.filter-btn'));
            filterProposals('all', document.querySelector('#allProposalsList').parentElement.querySelector('.filter-btn'));
        }
        
        // 제안 목록 렌더링
        function renderProposals() {
            filterProposals('all', document.querySelector('#proposalsTabList').parentElement.querySelector('.filter-btn'));
        }
        
        // 프로젝트 목록 렌더링
        function renderProjects() {
            filterProjects('all', document.querySelector('#projectsTabList').parentElement.querySelector('.filter-btn'));
        }
        
        // 리뷰 목록 렌더링
        function renderReviews() {
            const reviewsList = document.getElementById('reviewsList');
            reviewsList.innerHTML = mockData.reviews.length > 0
                ? mockData.reviews.map(r => createReviewCard(r)).join('')
                : '<div class="empty-state"><i class="fas fa-star"></i><p>받은 리뷰가 없습니다.</p></div>';
        }
        
        // 제안 상세 보기
        function showProposalDetail(proposalId) {
            const proposal = mockData.proposals.find(p => p.id === proposalId);
            if (!proposal) return;
            
            currentProposalId = proposalId;
            
            const modalBody = document.getElementById('proposalModalBody');
            const modalFooter = document.getElementById('proposalModalFooter');
            
            modalBody.innerHTML = `
                <p><strong>회사:</strong> ${proposal.company}</p>
                <p><strong>제목:</strong> ${proposal.title}</p>
                <p><strong>상세 내용:</strong></p>
                <p>${proposal.details}</p>
                <p><strong>제안일:</strong> ${proposal.date}</p>
            `;
            
            if (proposal.status === 'pending') {
                modalFooter.innerHTML = `
                    <button class="btn btn-danger" onclick="updateProposalStatus('rejected')">거절</button>
                    <button class="btn btn-success" onclick="updateProposalStatus('accepted')">수락</button>
                `;
            } else {
                modalFooter.innerHTML = `<button class="btn btn-secondary" onclick="closeProposalModal()">닫기</button>`;
            }
            
            document.getElementById('proposalModal').classList.add('active');
        }
        
        // 제안 상태 업데이트
        function updateProposalStatus(newStatus) {
            const proposal = mockData.proposals.find(p => p.id === currentProposalId);
            if (proposal) {
                proposal.status = newStatus;
                showToast(`제안을 ${newStatus === 'accepted' ? '수락' : '거절'}했습니다.`);
                closeProposalModal();
                
                // 모든 관련 뷰 업데이트
                renderDashboard();
                renderProposals();
            }
        }
        
        // 모달 닫기
        function closeProposalModal() {
            document.getElementById('proposalModal').classList.remove('active');
        }
        
        // 이벤트 리스너
        document.addEventListener('DOMContentLoaded', function() {
            // 초기 렌더링
            renderDashboard();
            renderProposals();
            renderProjects();
            renderReviews();
            
            // 탭 전환
            document.querySelectorAll('.tab-button').forEach(button => {
                button.addEventListener('click', function() {
                    showTab(this.dataset.tab);
                });
            });
            
            // 뒤로가기 버튼
            document.getElementById('backButton').addEventListener('click', function() {
                showToast('뒤로가기 기능은 구현 예정입니다.');
            });
            
            // 프로필 수정
            const editBtn = document.getElementById('editProfileBtn');
            const saveBtn = document.getElementById('saveProfileBtn');
            const formFields = document.querySelectorAll('#profileForm .form-control');
            
            editBtn.addEventListener('click', function() {
                formFields.forEach(field => field.disabled = false);
                editBtn.style.display = 'none';
                saveBtn.style.display = 'block';
            });
            
            saveBtn.addEventListener('click', function() {
                formFields.forEach(field => field.disabled = true);
                saveBtn.style.display = 'none';
                editBtn.style.display = 'block';
                showToast('프로필이 저장되었습니다.');
            });
            
            // 모달 닫기 버튼
            document.getElementById('closeProposalModal').addEventListener('click', closeProposalModal);
            
            // 모달 배경 클릭 시 닫기
            document.getElementById('proposalModal').addEventListener('click', function(e) {
                if (e.target === this) {
                    closeProposalModal();
                }
            });
        });
    </script>
